CREATE TABLE Users (
    username VARCHAR(30) PRIMARY KEY NOT NULL,
    password VARCHAR(30) NOT NULL,
    lesson INT NOT NULL
);

CREATE TABLE Lessons (
    id INT(6) unsigned AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    content TEXT NOT NULL,
    quiz INT(30) NOT NULL
);

CREATE TABLE Quizes(
    id INT NOT NULL, 
    content TEXT NOT NULL
);
